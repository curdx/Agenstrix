          { regex: /sk-ant-[A-Za-z0-9_\-]{20,}/g,       label: "ANTHROPIC-KEY" },
          { regex: /ghp_[A-Za-z0-9]{36}/g,               label: "GITHUB-TOKEN" },

        ];
        
        const encoder = new TextEncoder();
        const decoder = new TextDecoder("utf-8", { fatal: false });
        
        export function redactString(s: string): string {
          let out = s;
          for (const { regex, label } of PATTERNS) {
            out = out.replace(regex, `[REDACTED-${label}]`);
          }
          return out;
        }
        
        export function redactChunk(chunk: Uint8Array): Uint8Array {
          // Fast-path: no ESC, no 'sk-', no 'ghp_', no 'AKIA' in raw bytes → skip
          // (heuristic byte scan; avoids regex when chunk is pure terminal output)
          if (!hasSecretPrefix(chunk)) return chunk;
          const str = decoder.decode(chunk);
          const redacted = redactString(str);
          if (redacted === str) return chunk;
          return encoder.encode(redacted);
        }
        
        function hasSecretPrefix(chunk: Uint8Array): boolean {
          // Look for "sk-", "ghp", "AKIA" byte patterns — case-sensitive for speed
          for (let i = 0; i < chunk.length - 3; i++) {
            if (chunk[i] === 0x73 /* s */ && chunk[i+1] === 0x6B /* k */ && chunk[i+2] === 0x2D /* - */) return true;
            if (chunk[i] === 0x67 /* g */ && chunk[i+1] === 0x68 /* h */ && chunk[i+2] === 0x70 /* p */) return true;
            if (chunk[i] === 0x41 /* A */ && chunk[i+1] === 0x4B /* K */ && chunk[i+2] === 0x49 /* I */ && chunk[i+3] === 0x41 /* A */) return true;
          }
          return false;
        }
